SOURCE
https://www.qwiklabs.com/focuses/1095

REPO
git clone https://github.com/googlecodelabs/tensorflow-for-poets-2

FLOWER_IMAGES
curl -o flower_photos.tgz "http://download.tensorflow.org/example_images/flower_photos.tgz"
tar zxvf ./flower_photos.tgz
- move to tf_files

TENSORBOARD
tensorboard --logdir tf_files/training_summaries &

RETRAIN scripts
$ARCHITECHTURE="mobilenet_0.50_224" or "inception_v3"
- POWERSHELL
python -m scripts.retrain `
  --bottleneck_dir=tf_files/bottlenecks `
  --how_many_training_steps=500 `
  --model_dir=tf_files/models/ `
  --summaries_dir=tf_files/training_summaries/$ARCHITECHTURE `
  --output_graph=tf_files/retrained_graph.pb `
  --output_labels=tf_files/retrained_labels.txt `
  --architecture=$ARCHITECHTURE `
  --image_dir=tf_files/flower_photos

- BASH
python -m scripts.retrain \
  --bottleneck_dir=tf_files/bottlenecks \
  --how_many_training_steps=500 \
  --model_dir=tf_files/models/ \
  --summaries_dir=tf_files/training_summaries/"${ARCHITECHTURE}" \
  --output_graph=tf_files/retrained_graph.pb \
  --output_labels=tf_files/retrained_labels.txt \
  --architecture="${ARCHITECHTURE}" \
  --image_dir=tf_files/flower_photos
